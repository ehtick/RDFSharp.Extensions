/* This is the official test suite of RDFMySQLStore. It uses a test database on localhost with test:test */

using RDFSharp.Extensions.MySQL;
using RDFSharp.Model;
using RDFSharp.Store;
using Xunit;

//Setup
RDFMySQLStore mySqlStore = new RDFMySQLStore("Server=localhost;Port=3306;Database=SampleDatabase;Uid=test;Pwd=test;Charset=utf8mb4;");
Assert.NotNull(mySqlStore);
Assert.Equal("MYSQL", mySqlStore.StoreType);

//Clear
if (new Random().NextDouble() >= 0.50)
    await mySqlStore.ClearQuadruplesAsync();
else
    mySqlStore.ClearQuadruples();

Assert.Equal(0, mySqlStore.QuadruplesCount);
Assert.Equal(0, await mySqlStore.QuadruplesCountAsync);

//Load
for (int i = 1; i <= 250; i++)
{
    mySqlStore.AddQuadruple(new RDFQuadruple(
            new RDFContext(new Uri($"ex:ctx{i}")),
            new RDFResource($"ex:subj{i}"),
            new RDFResource($"ex:pred{i}"),
            new RDFResource($"ex:obj{i}")));
    await mySqlStore.AddQuadrupleAsync(new RDFQuadruple(
        new RDFContext(new Uri($"ex:ctx{i}")),
        new RDFResource($"ex:subj{i}"),
        new RDFResource($"ex:pred{i}"),
        new RDFPlainLiteral($"{i}")));
}
Assert.Equal(500, mySqlStore.QuadruplesCount);
Assert.Equal(500, await mySqlStore.QuadruplesCountAsync);
mySqlStore.AddQuadruple(new RDFQuadruple(
    new RDFContext(new Uri("ex:ctx200")),
    new RDFResource("ex:subj200"),
    new RDFResource("ex:pred200"),
    new RDFResource("ex:obj200"))); //Already existing
Assert.Equal(500, mySqlStore.QuadruplesCount);
Assert.Equal(500, await mySqlStore.QuadruplesCountAsync);
RDFGraph graph1 = new RDFGraph([
    new RDFTriple(new RDFResource("ex:subj251"), new RDFResource("ex:pred251"), new RDFPlainLiteral("251")),
    new RDFTriple(new RDFResource("ex:subj251"), new RDFResource("ex:pred251"), new RDFResource("ex:obj251"))]);
mySqlStore.MergeGraph(graph1);
Assert.Equal(502, mySqlStore.QuadruplesCount);
Assert.Equal(502, await mySqlStore.QuadruplesCountAsync);
RDFGraph graph2 = new RDFGraph([
    new RDFTriple(new RDFResource("ex:subj252"), new RDFResource("ex:pred252"), new RDFPlainLiteral("252")),
    new RDFTriple(new RDFResource("ex:subj252"), new RDFResource("ex:pred252"), new RDFResource("ex:obj252"))]);
await mySqlStore.MergeGraphAsync(graph2);
Assert.Equal(504, mySqlStore.QuadruplesCount);
Assert.Equal(504, await mySqlStore.QuadruplesCountAsync);

//Remove
mySqlStore.RemoveQuadruple(new RDFQuadruple(
    new RDFContext(),

    new RDFResource("ex:pred252"),
    new RDFResource("ex:obj252")));

Assert.Equal(503, await mySqlStore.QuadruplesCountAsync);

    new RDFContext(),
    new RDFResource("ex:subj254"),

    new RDFResource("ex:obj254"))); //unexisting

Assert.Equal(503, await mySqlStore.QuadruplesCountAsync);
await mySqlStore.RemoveQuadruplesAsync(c:new RDFContext());

Assert.Equal(500, await mySqlStore.QuadruplesCountAsync);

Assert.Equal(495, await mySqlStore.QuadruplesCountAsync);

Assert.Equal(494, await mySqlStore.QuadruplesCountAsync);
await mySqlStore.RemoveQuadruplesAsync(c:new RDFContext(new Uri("ex:ctx246")),s:new RDFResource("ex:subj246"));

await mySqlStore.RemoveQuadruplesAsync(c:new RDFContext(new Uri("ex:ctx244")),o:new RDFResource("ex:obj244"));

Assert.Equal(474, mySqlStore.QuadruplesCount);
Assert.Equal(474, await mySqlStore.QuadruplesCountAsync);

//Query
Assert.True(mySqlStore.ContainsQuadruple(new RDFQuadruple(
    new RDFContext(new Uri("ex:ctx200")),
    new RDFResource("ex:subj200"),
    new RDFResource("ex:pred200"),
    new RDFResource("ex:obj200"))));
Assert.True(await mySqlStore.ContainsQuadrupleAsync(new RDFQuadruple(
    new RDFContext(new Uri("ex:ctx200")),
    new RDFResource("ex:subj200"),
    new RDFResource("ex:pred200"),
    new RDFResource("ex:obj200"))));
Assert.False(mySqlStore.ContainsQuadruple(new RDFQuadruple(
    new RDFContext(new Uri("ex:ctx2007787")),
    new RDFResource("ex:subj200"),
    new RDFResource("ex:pred200"),
    new RDFResource("ex:obj200"))));
Assert.False(await mySqlStore.ContainsQuadrupleAsync(new RDFQuadruple(
    new RDFContext(new Uri("ex:ctx2007787")),
    new RDFResource("ex:subj200"),
    new RDFResource("ex:pred200"),
    new RDFResource("ex:obj200"))));
Assert.Equal(2, mySqlStore.SelectQuadruples(c: new RDFContext(new Uri("ex:ctx200"))).Count);
Assert.Equal(2, mySqlStore.SelectQuadruples(s: new RDFResource("ex:subj200")).Count);
Assert.Equal(2, mySqlStore.SelectQuadruples(p: new RDFResource("ex:pred200")).Count);
Assert.Equal(1, mySqlStore.SelectQuadruples(o: new RDFResource("ex:obj200")).Count);
Assert.Equal(1, mySqlStore.SelectQuadruples(l: new RDFPlainLiteral("200")).Count);
Assert.Equal(2, mySqlStore.SelectQuadruples(c: new RDFContext(new Uri("ex:ctx200")), s: new RDFResource("ex:subj200")).Count);
Assert.Equal(2, mySqlStore.SelectQuadruples(c: new RDFContext(new Uri("ex:ctx200")), p: new RDFResource("ex:pred200")).Count);
Assert.Equal(1, mySqlStore.SelectQuadruples(c: new RDFContext(new Uri("ex:ctx200")), o: new RDFResource("ex:obj200")).Count);
Assert.Equal(1, mySqlStore.SelectQuadruples(c: new RDFContext(new Uri("ex:ctx200")), l: new RDFPlainLiteral("200")).Count);
Assert.Equal(2, mySqlStore.SelectQuadruples(c: new RDFContext(new Uri("ex:ctx200")), s: new RDFResource("ex:subj200"), p: new RDFResource("ex:pred200")).Count);
Assert.Equal(1, mySqlStore.SelectQuadruples(c: new RDFContext(new Uri("ex:ctx200")), s: new RDFResource("ex:subj200"), o: new RDFResource("ex:obj200")).Count);
Assert.Equal(1, mySqlStore.SelectQuadruples(c: new RDFContext(new Uri("ex:ctx200")), s: new RDFResource("ex:subj200"), l: new RDFPlainLiteral("200")).Count);
Assert.Equal(1, mySqlStore.SelectQuadruples(c: new RDFContext(new Uri("ex:ctx200")), s: new RDFResource("ex:subj200"), p: new RDFResource("ex:pred200"), o: new RDFResource("ex:obj200")).Count);
Assert.Equal(1, mySqlStore.SelectQuadruples(c: new RDFContext(new Uri("ex:ctx200")), s: new RDFResource("ex:subj200"), p: new RDFResource("ex:pred200"), l: new RDFPlainLiteral("200")).Count);
Assert.Equal(2, mySqlStore.SelectQuadruples(s: new RDFResource("ex:subj200"), p: new RDFResource("ex:pred200")).Count);
Assert.Equal(1, mySqlStore.SelectQuadruples(s: new RDFResource("ex:subj200"), o: new RDFResource("ex:obj200")).Count);
Assert.Equal(1, mySqlStore.SelectQuadruples(s: new RDFResource("ex:subj200"), l: new RDFPlainLiteral("200")).Count);
Assert.Equal(1, mySqlStore.SelectQuadruples(s: new RDFResource("ex:subj200"), p: new RDFResource("ex:pred200"), o: new RDFResource("ex:obj200")).Count);
Assert.Equal(1, mySqlStore.SelectQuadruples(s: new RDFResource("ex:subj200"), p: new RDFResource("ex:pred200"), l: new RDFPlainLiteral("200")).Count);
Assert.Equal(1, mySqlStore.SelectQuadruples(p: new RDFResource("ex:pred200"), o: new RDFResource("ex:obj200")).Count);
Assert.Equal(1, mySqlStore.SelectQuadruples(p: new RDFResource("ex:pred200"), l: new RDFPlainLiteral("200")).Count);
Assert.Equal(0, mySqlStore.SelectQuadruples(p: new RDFResource("ex:pred200786537"), l: new RDFPlainLiteral("200")).Count); //unexisting

//Query (Full Selectors + Accessors)
Assert.True(mySqlStore.SelectQuadruples().Count > 100);
Assert.Equal(1, mySqlStore[p: new RDFResource("ex:pred200"), l: new RDFPlainLiteral("200")].QuadruplesCount);
Assert.True(mySqlStore[null, null, null, null, null].QuadruplesCount > 100);
await mySqlStore.RemoveQuadruplesAsync();
Assert.Equal(0, mySqlStore.QuadruplesCount);
Assert.Equal(0, await mySqlStore.QuadruplesCountAsync);
Assert.True(mySqlStore.SelectQuadruples().Count == 0);
Assert.True(mySqlStore[null, null, null, null, null].QuadruplesCount == 0);
