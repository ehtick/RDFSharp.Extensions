/* This is the official test suite of RDFFirebirdStore. It uses a temporary Linux database (/tmp) on localhost with SYSDBA:sysdba */

using RDFSharp.Extensions.Firebird;
using RDFSharp.Model;
using RDFSharp.Store;
using Xunit;

//Setup
RDFFirebirdStore fbStore = new RDFFirebirdStore("User=SYSDBA;Password=sysdba;Database=/tmp/SampleDatabase.fdb;DataSource=localhost;Port=3050;Charset=NONE;Dialect=3;ServerType=0;");
Assert.NotNull(fbStore);
Assert.Equal("FIREBIRD", fbStore.StoreType);

//Clear
if (new Random().NextDouble() >= 0.50)
    await fbStore.ClearQuadruplesAsync();
else
    fbStore.ClearQuadruples();

Assert.Equal(0, fbStore.QuadruplesCount);
Assert.Equal(0, await fbStore.QuadruplesCountAsync);

//Load
for (int i = 1; i <= 250; i++)
{
    fbStore.AddQuadruple(new RDFQuadruple(
            new RDFContext(new Uri($"ex:ctx{i}")),
            new RDFResource($"ex:subj{i}"),
            new RDFResource($"ex:pred{i}"),
            new RDFResource($"ex:obj{i}")));
    await fbStore.AddQuadrupleAsync(new RDFQuadruple(
        new RDFContext(new Uri($"ex:ctx{i}")),
        new RDFResource($"ex:subj{i}"),
        new RDFResource($"ex:pred{i}"),
        new RDFPlainLiteral($"{i}")));
}
Assert.Equal(500, fbStore.QuadruplesCount);
Assert.Equal(500, await fbStore.QuadruplesCountAsync);
fbStore.AddQuadruple(new RDFQuadruple(
    new RDFContext(new Uri("ex:ctx200")),
    new RDFResource("ex:subj200"),
    new RDFResource("ex:pred200"),
    new RDFResource("ex:obj200"))); //Already existing
Assert.Equal(500, fbStore.QuadruplesCount);
Assert.Equal(500, await fbStore.QuadruplesCountAsync);
RDFGraph graph1 = new RDFGraph([
    new RDFTriple(new RDFResource("ex:subj251"), new RDFResource("ex:pred251"), new RDFPlainLiteral("251")),
    new RDFTriple(new RDFResource("ex:subj251"), new RDFResource("ex:pred251"), new RDFResource("ex:obj251"))]);
fbStore.MergeGraph(graph1);
Assert.Equal(502, fbStore.QuadruplesCount);
Assert.Equal(502, await fbStore.QuadruplesCountAsync);
RDFGraph graph2 = new RDFGraph([
    new RDFTriple(new RDFResource("ex:subj252"), new RDFResource("ex:pred252"), new RDFPlainLiteral("252")),
    new RDFTriple(new RDFResource("ex:subj252"), new RDFResource("ex:pred252"), new RDFResource("ex:obj252"))]);
await fbStore.MergeGraphAsync(graph2);
Assert.Equal(504, fbStore.QuadruplesCount);
Assert.Equal(504, await fbStore.QuadruplesCountAsync);

//Remove
fbStore.RemoveQuadruple(new RDFQuadruple(
    new RDFContext(),
    new RDFResource("ex:subj252"),
    new RDFResource("ex:pred252"),
    new RDFResource("ex:obj252")));
Assert.Equal(503, fbStore.QuadruplesCount);
Assert.Equal(503, await fbStore.QuadruplesCountAsync);
fbStore.RemoveQuadruple(new RDFQuadruple(
    new RDFContext(),
    new RDFResource("ex:subj254"),
    new RDFResource("ex:pred254"),
    new RDFResource("ex:obj254"))); //unexisting
Assert.Equal(503, fbStore.QuadruplesCount);
Assert.Equal(503, await fbStore.QuadruplesCountAsync);
await fbStore.RemoveQuadruplesAsync(c:new RDFContext());
Assert.Equal(500, fbStore.QuadruplesCount);
Assert.Equal(500, await fbStore.QuadruplesCountAsync);
await fbStore.RemoveQuadruplesAsync(s:new RDFResource("ex:subj250"));
Assert.Equal(498, fbStore.QuadruplesCount);
Assert.Equal(498, await fbStore.QuadruplesCountAsync);
await fbStore.RemoveQuadruplesAsync(p:new RDFResource("ex:pred249"));
Assert.Equal(496, fbStore.QuadruplesCount);
Assert.Equal(496, await fbStore.QuadruplesCountAsync);
await fbStore.RemoveQuadruplesAsync(o:new RDFResource("ex:obj248"));
Assert.Equal(495, fbStore.QuadruplesCount);
Assert.Equal(495, await fbStore.QuadruplesCountAsync);
await fbStore.RemoveQuadruplesAsync(l:new RDFPlainLiteral("247"));
Assert.Equal(494, fbStore.QuadruplesCount);
Assert.Equal(494, await fbStore.QuadruplesCountAsync);
await fbStore.RemoveQuadruplesAsync(c:new RDFContext(new Uri("ex:ctx246")),s:new RDFResource("ex:subj246"));
Assert.Equal(492, fbStore.QuadruplesCount);
Assert.Equal(492, await fbStore.QuadruplesCountAsync);
await fbStore.RemoveQuadruplesAsync(c:new RDFContext(new Uri("ex:ctx245")),p:new RDFResource("ex:pred245"));
Assert.Equal(490, fbStore.QuadruplesCount);
Assert.Equal(490, await fbStore.QuadruplesCountAsync);
await fbStore.RemoveQuadruplesAsync(c:new RDFContext(new Uri("ex:ctx244")),o:new RDFResource("ex:obj244"));
Assert.Equal(489, fbStore.QuadruplesCount);
Assert.Equal(489, await fbStore.QuadruplesCountAsync);
await fbStore.RemoveQuadruplesAsync(c:new RDFContext(new Uri("ex:ctx243")),l:new RDFPlainLiteral("243"));
Assert.Equal(488, fbStore.QuadruplesCount);
Assert.Equal(488, await fbStore.QuadruplesCountAsync);
await fbStore.RemoveQuadruplesAsync(c:new RDFContext(new Uri("ex:ctx242")),s:new RDFResource("ex:subj242"),p:new RDFResource("ex:pred242"));
Assert.Equal(486, fbStore.QuadruplesCount);
Assert.Equal(486, await fbStore.QuadruplesCountAsync);
await fbStore.RemoveQuadruplesAsync(c:new RDFContext(new Uri("ex:ctx241")),s:new RDFResource("ex:subj241"),o:new RDFResource("ex:obj241"));
Assert.Equal(485, fbStore.QuadruplesCount);
Assert.Equal(485, await fbStore.QuadruplesCountAsync);
await fbStore.RemoveQuadruplesAsync(c:new RDFContext(new Uri("ex:ctx240")),s:new RDFResource("ex:subj240"),l:new RDFPlainLiteral("240"));
Assert.Equal(484, fbStore.QuadruplesCount);
Assert.Equal(484, await fbStore.QuadruplesCountAsync);
await fbStore.RemoveQuadruplesAsync(c:new RDFContext(new Uri("ex:ctx239")),s:new RDFResource("ex:subj239"),p:new RDFResource("ex:pred239"),o:new RDFResource("ex:obj239"));
Assert.Equal(483, fbStore.QuadruplesCount);
Assert.Equal(483, await fbStore.QuadruplesCountAsync);
await fbStore.RemoveQuadruplesAsync(c:new RDFContext(new Uri("ex:ctx238")),s:new RDFResource("ex:subj238"),p:new RDFResource("ex:pred238"),l:new RDFPlainLiteral("238"));
Assert.Equal(482, fbStore.QuadruplesCount);
Assert.Equal(482, await fbStore.QuadruplesCountAsync);
await fbStore.RemoveQuadruplesAsync(s:new RDFResource("ex:subj237"),p:new RDFResource("ex:pred237"));
Assert.Equal(480, fbStore.QuadruplesCount);
Assert.Equal(480, await fbStore.QuadruplesCountAsync);
await fbStore.RemoveQuadruplesAsync(s:new RDFResource("ex:subj236"),o:new RDFResource("ex:obj236"));
Assert.Equal(479, fbStore.QuadruplesCount);
Assert.Equal(479, await fbStore.QuadruplesCountAsync);
await fbStore.RemoveQuadruplesAsync(s:new RDFResource("ex:subj235"),l:new RDFPlainLiteral("235"));
Assert.Equal(478, fbStore.QuadruplesCount);
Assert.Equal(478, await fbStore.QuadruplesCountAsync);
await fbStore.RemoveQuadruplesAsync(s:new RDFResource("ex:subj234"),p:new RDFResource("ex:pred234"),o:new RDFResource("ex:obj234"));

Assert.Equal(477, await fbStore.QuadruplesCountAsync);
await fbStore.RemoveQuadruplesAsync(s:new RDFResource("ex:subj233"),p:new RDFResource("ex:pred233"),l:new RDFPlainLiteral("233"));

await fbStore.RemoveQuadruplesAsync(p:new RDFResource("ex:pred232"),o:new RDFResource("ex:obj232"));
Assert.Equal(475, fbStore.QuadruplesCount);
Assert.Equal(475, await fbStore.QuadruplesCountAsync);
await fbStore.RemoveQuadruplesAsync(p:new RDFResource("ex:pred231"),l:new RDFPlainLiteral("231"));
Assert.Equal(474, fbStore.QuadruplesCount);
Assert.Equal(474, await fbStore.QuadruplesCountAsync);
await fbStore.RemoveQuadruplesAsync(p:new RDFResource("ex:pred231"),l:new RDFPlainLiteral("231")); //no more existing
Assert.Equal(474, fbStore.QuadruplesCount);
Assert.Equal(474, await fbStore.QuadruplesCountAsync);

//Query
Assert.True(fbStore.ContainsQuadruple(new RDFQuadruple(
    new RDFContext(new Uri("ex:ctx200")),
    new RDFResource("ex:subj200"),
    new RDFResource("ex:pred200"),
    new RDFResource("ex:obj200"))));
Assert.True(await fbStore.ContainsQuadrupleAsync(new RDFQuadruple(
    new RDFContext(new Uri("ex:ctx200")),
    new RDFResource("ex:subj200"),
    new RDFResource("ex:pred200"),
    new RDFResource("ex:obj200"))));
Assert.False(fbStore.ContainsQuadruple(new RDFQuadruple(
    new RDFContext(new Uri("ex:ctx2007787")),
    new RDFResource("ex:subj200"),
    new RDFResource("ex:pred200"),
    new RDFResource("ex:obj200"))));
Assert.False(await fbStore.ContainsQuadrupleAsync(new RDFQuadruple(
    new RDFContext(new Uri("ex:ctx2007787")),
    new RDFResource("ex:subj200"),
    new RDFResource("ex:pred200"),
    new RDFResource("ex:obj200"))));
Assert.Equal(2, fbStore.SelectQuadruples(c: new RDFContext(new Uri("ex:ctx200"))).Count);
Assert.Equal(2, fbStore.SelectQuadruples(s: new RDFResource("ex:subj200")).Count);
Assert.Equal(2, fbStore.SelectQuadruples(p: new RDFResource("ex:pred200")).Count);
Assert.Equal(1, fbStore.SelectQuadruples(o: new RDFResource("ex:obj200")).Count);
Assert.Equal(1, fbStore.SelectQuadruples(l: new RDFPlainLiteral("200")).Count);
Assert.Equal(2, fbStore.SelectQuadruples(c: new RDFContext(new Uri("ex:ctx200")), s: new RDFResource("ex:subj200")).Count);
Assert.Equal(2, fbStore.SelectQuadruples(c: new RDFContext(new Uri("ex:ctx200")), p: new RDFResource("ex:pred200")).Count);
Assert.Equal(1, fbStore.SelectQuadruples(c: new RDFContext(new Uri("ex:ctx200")), o: new RDFResource("ex:obj200")).Count);
Assert.Equal(1, fbStore.SelectQuadruples(c: new RDFContext(new Uri("ex:ctx200")), l: new RDFPlainLiteral("200")).Count);
Assert.Equal(2, fbStore.SelectQuadruples(c: new RDFContext(new Uri("ex:ctx200")), s: new RDFResource("ex:subj200"), p: new RDFResource("ex:pred200")).Count);
Assert.Equal(1, fbStore.SelectQuadruples(c: new RDFContext(new Uri("ex:ctx200")), s: new RDFResource("ex:subj200"), o: new RDFResource("ex:obj200")).Count);
Assert.Equal(1, fbStore.SelectQuadruples(c: new RDFContext(new Uri("ex:ctx200")), s: new RDFResource("ex:subj200"), l: new RDFPlainLiteral("200")).Count);
Assert.Equal(1, fbStore.SelectQuadruples(c: new RDFContext(new Uri("ex:ctx200")), s: new RDFResource("ex:subj200"), p: new RDFResource("ex:pred200"), o: new RDFResource("ex:obj200")).Count);
Assert.Equal(1, fbStore.SelectQuadruples(c: new RDFContext(new Uri("ex:ctx200")), s: new RDFResource("ex:subj200"), p: new RDFResource("ex:pred200"), l: new RDFPlainLiteral("200")).Count);
Assert.Equal(2, fbStore.SelectQuadruples(s: new RDFResource("ex:subj200"), p: new RDFResource("ex:pred200")).Count);
Assert.Equal(1, fbStore.SelectQuadruples(s: new RDFResource("ex:subj200"), o: new RDFResource("ex:obj200")).Count);
Assert.Equal(1, fbStore.SelectQuadruples(s: new RDFResource("ex:subj200"), l: new RDFPlainLiteral("200")).Count);
Assert.Equal(1, fbStore.SelectQuadruples(s: new RDFResource("ex:subj200"), p: new RDFResource("ex:pred200"), o: new RDFResource("ex:obj200")).Count);
Assert.Equal(1, fbStore.SelectQuadruples(s: new RDFResource("ex:subj200"), p: new RDFResource("ex:pred200"), l: new RDFPlainLiteral("200")).Count);
Assert.Equal(1, fbStore.SelectQuadruples(p: new RDFResource("ex:pred200"), o: new RDFResource("ex:obj200")).Count);
Assert.Equal(1, fbStore.SelectQuadruples(p: new RDFResource("ex:pred200"), l: new RDFPlainLiteral("200")).Count);
Assert.Equal(0, fbStore.SelectQuadruples(p: new RDFResource("ex:pred200786537"), l: new RDFPlainLiteral("200")).Count); //unexisting

//Query (Full Selectors + Accessors)
Assert.True(fbStore.SelectQuadruples().Count > 100);
Assert.Equal(1, fbStore[p: new RDFResource("ex:pred200"), l: new RDFPlainLiteral("200")].QuadruplesCount);
Assert.True(fbStore[null, null, null, null, null].QuadruplesCount > 100);
await fbStore.RemoveQuadruplesAsync();
Assert.Equal(0, fbStore.QuadruplesCount);
Assert.Equal(0, await fbStore.QuadruplesCountAsync);
Assert.True(fbStore.SelectQuadruples().Count == 0);
Assert.True(fbStore[null, null, null, null, null].QuadruplesCount == 0);
